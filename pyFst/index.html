def x(inp: str, out: str):
    """One-symbol transducer."""
    return pynini.cross(inp, out)




def apply_fst(fst, text: str):
    """
    Apply FST to text and return the output string.
    Returns None if the input is rejected.
    """
    lattice = pynini.accep(text) @ fst

        return pynini.shortestpath(lattice).string()
    except pywrapfst.FstOpError:
        return None

from pathlib import Path
import string
import subprocess

import pandas as pd
import pynini
import pywrapfst
from pynini.lib import pynutil
from IPython.display import Image, display


# ----------------------------
# Helpers
# ----------------------------

def accep(s: str):
    return pynini.accep(s)


def word(prefix: str, stem: str, suffix: str):
    """
    Build lexical forms while omitting empty morphemes.

    Examples:
      word("",   "run",   "")     -> run
      word("un", "happy", "")     -> un+happy
      word("",   "run",   "er")   -> run+er
      word("un", "happy", "ness") -> un+happy+ness
    """
    parts = [part for part in [prefix, stem, suffix] if part != ""]
    return pynini.accep("+".join(parts))


def byte_symbol_table(chars: str):
    """
    Pynini's default ASCII labels are byte integers.
    This table makes Graphviz show letters instead of numbers.
    """
    syms = pywrapfst.SymbolTable()
    syms.add_symbol("<eps>", 0)
    for ch in sorted(set(chars)):
        syms.add_symbol(ch, ord(ch))
    return syms


def one_best(fst, text: str):
    """
    Apply FST to text and return the best output string.
    Returns None if the input is rejected.
    """
    lattice = pynini.compose(accep(text), fst)

    # No valid path.
    if lattice.start() == pywrapfst.NO_STATE_ID:
        return None

    try:
        path = pynini.shortestpath(lattice, nshortest=1, unique=True).optimize()

        # Critical fix:
        # convert the relation into its output string language.
        path.project("output")
        path.rmepsilon()

        return path.string()
    except pywrapfst.FstOpError:
        return None


# ----------------------------
# 1. Lexicon / morphotactics
# ----------------------------

LEXICON = pynini.union(
    word("",   "happy", ""),       # happy+er
    word("un", "happy", ""),       # un+happy
    word("",   "happy", "er"),     # un+happy+er
    word("un", "happy", "er"),     # un+happy+er
    word("",   "happy", "est"),    # happy+est
    word("un", "happy", "est"),    # un+happy+est
    word("",   "happy", "ness"),   # happy+ness
    word("un", "happy", "ness"),   # un+happy+ness
    word("",   "run",   ""),       # run
    word("",   "run",   "er"),     # run+er
    word("",   "run",   "ing"),    # run+ing
    word("re", "run",   "ing"),    # re+run+ing
    word("",   "jump",  ""),       # jump
    word("",   "jump",  "ed"),     # jump+ed
    word("",   "jump",  "ing"),    # jump+ing
    word("",   "jump",  "er"),     # jump+er/N
    word("",   "jump",  "y"),     # jump+y/ADJ
    word("",   "jumpy",  "er"),    # jumpy+ier/ADJ
    word("",   "jumpy",  "est"),   # jumpy+iest/ADJ
).optimize()


# ----------------------------
# 2. Spelling rules
# ----------------------------

ALPHABET = string.ascii_lowercase + "+"
SIGMA_STAR = pynini.union(*ALPHABET).closure().optimize()

PLUS = accep("+")
ING = accep("ing")
NESS = accep("ness")
ER = accep("er")


# happy+ness -> happi+ness
y_to_i_before_suffix = pynini.cdrewrite(
    pynini.cross("y", "i"),
    "",
    PLUS + (NESS | ER),
    SIGMA_STAR,
).optimize()


# run+er -> runn+er
double_n_before_er = pynini.cdrewrite(
    pynutil.insert("n"),
    accep("n"),
    PLUS + ER,
    SIGMA_STAR,
).optimize()

# run+ing -> runn+ing
double_n_before_ing = pynini.cdrewrite(
    pynutil.insert("n"),
    accep("n"),
    PLUS + ING,
    SIGMA_STAR,
).optimize()


# jump+ing -> jumping
# re+runn+ing -> rerunning
# un+happi+ness -> unhappiness
delete_boundaries = pynini.cdrewrite(
    pynutil.delete("+"),
    "",
    "",
    SIGMA_STAR,
).optimize()

# 3. Generator and analyzer
# ----------------------------

GENERATOR = (

    @ double_n_before_er
    @ double_n_before_ing
    @ delete_boundaries

The above analyzer is good at breaking down surface forms into their underlying lexical morphemes. However we should get a more detailed analysis of the morphemes, e.g. happy+ness -> happy/STEM+ness/NMLZ This will not only a more detailed analysis but also simplify generation since we can have a separate FST that maps from the compact feature specifications to the full tagged analyses, and then use the same spelling rules for generation and analysis.

How this sample works:

Overview

The sample separates morphological analysis into two layers. The first layer maps between surface spellings and underlying lexical morphemes: for example, happiness is analyzed as happy+ness, and rerunning as re+run+ing. This captures the hidden morpheme boundaries and reverses spelling alternations such as y → i in happy → happiness or consonant doubling in run → running.

The second layer enriches those morphemes with grammatical information. Instead of stopping at happy+ness, the analyzer can return happy/STEM<ADJ>+ness/NMLZ<N>, making explicit that happy is an adjectival stem and -ness is a nominalizing suffix that derives a noun. This richer tagged representation is also useful for generation: a compact feature request such as happy/ADJ<NMLZ> can first be expanded into the full tagged analysis, then passed through the same lexical and spelling-rule pipeline that the analyzer uses in reverse. This keeps the system modular: features choose the morphology; spelling rules realize it.

def one_best(fst, text: str):
    """
    Apply FST to text and return the best output string.
    Returns None if the input is rejected.
    """
    lattice = pynini.compose(accep(text), fst)

    if lattice.start() == pywrapfst.NO_STATE_ID:

    try:
        path = pynini.shortestpath(lattice, nshortest=1, unique=True).optimize()
        path.project("output")
        path.rmepsilon()
        return path.string()
    except pywrapfst.FstOpError:
